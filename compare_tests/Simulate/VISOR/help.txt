Options:
	-d DIMENSIONS, --dimensions=DIMENSIONS
		TSV file with chromosome dimensions (as result from 'cut -f1,2 genome.fa.fai') [required]

	-n NUMBER, --number=NUMBER
		number of intervals [100]

	-l LENGTH, --length=LENGTH
		mean intervals length [50000]

	-s STANDARDDEV, --standarddev=STANDARDDEV
		standard deviation for intervals length [0]

	-x EXCLUDE, --exclude=EXCLUDE
		exclude regions in BED [optional]

	-v VARIANTS, --variants=VARIANTS
		variants types (-v 'deletion,tandem duplication,inversion') [required]

	-r RATIO, --ratio=RATIO
		variants proportions (-r '30:30:40')) [required]

	-i IDHAPLO, --idhaplo=IDHAPLO
		haplotype number [1]

	-m MICROSATELLITES, --microsatellites=MICROSATELLITES
		BED file with microsatellites (ucsc format) [optional]

	-g GENOME, --genome=GENOME
		FASTA genome [optional]
